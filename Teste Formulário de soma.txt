<script>
    const firstDiv = document.querySelector('.first-step');
    const secondDiv = document.querySelector('.second-step');
    const finalDiv = document.querySelector('.final-step');


    function go(currentStep,nextStep)
    {
        let dNone, dBlock;
        if(currentStep == 1)
        {
            dNone = firstDiv;
        }

        else if (currentStep == 2)
        {
            dNone = secondDiv;
        }
        else
        {
            dNone = finalDiv;
        }

        dNone.style.display = 'none';

        if(nextStep == 1)
        {
            dBlock = firstDiv;
        }
        else
        {
            dBlock = finalDiv;
        }
        dBlock.style.display = 'block';
    }
    function validate()
    {
        const valor1 = document.getElementById('valor1');
        const valor2 = document.getElementById('valor2');
        valor1.style.border = 'none';
        valor2.style.border = 'none';
        if(!valor1.value || !valor2.value)
        {
            if(!valor1.value && !valor2.value)
            {
                valor1.style.border = '1px solid red';
                valor2.style.border = '1px solid red';
            }
            else if(!valor1.value)
            {
                valor1.style.border = '1px solid red';
            }
            else
            {
                valor2.style.border = '1px solid red';
            }
        }
        else
        {
            let soma = valor1.value + valor2.value;
            const result = document.getElementById('resultado')
            {
                console.log(soma);
                result.innerHTML = soma
            }
        }
        go(2,3);
        function somar()
        {
            var n1 = parseInt(document.getElementById('n1').value, 10);
            var n2 = parseInt(document.getElementById('n2').value, 10);
            document.getElementById('resultado').innerHTML = n1 + n2;
        }
    }
